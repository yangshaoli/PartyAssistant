1. createParty
	url:"/a/parties/createparty/"
	method:POST
	input:{
		"receivers":Array of Client Dict,
		"content":String of SMS/Email Content,
		"subject":String of Email Subject;sms will be blank,
		"_isapplytips":BOOL/Integer(Transfer to BOOL),
		"_issendbyself":BOOL/Integer(Transfer to BOOL),
		"msgType":String of "SMS" or "Email",
		"starttime":Datetime Type,
		"location":String,
		"description":String,
		"peopleMaximum":Int,
		"uID":Int,
		"addressType":"iphone" or "android"
		}
	output:Success- {
			"status":'ok',
			"description":"ok",
			"data":{}
			}
		Error - {
			"status":'error',
			"description":Error Msg,
			"data":{}
			}
	ClientDict:{
			"cName":String,
			"cVal":Email/Phone Num
			"cId":Integer(iphone中Client的Id，需记录其ID，存在位置："iPhone")
			}
