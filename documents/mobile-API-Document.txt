	input:{

			"description":"ok",

                		'partyId':Integer,

                		},

        	'android_version':ANDROID_APP_VERSION(String),
			}

			"status":'error',
			"description":Error Msg,
			"data":{},
			'iphone_version':IPHONE_APP_VERSION(String),
            'android_version':ANDROID_APP_VERSION(String),
			}
	ClientDict:{
			"cName":String,
			"cVal":Email/Phone Num
			"cId":Integer(iphone中Client的Id，需记录其ID，存在位置："iPhone")
			}
